Binary Search C 
// Binary search in sorted array
int binarySearch(int arr[], 
    int n, int target) {
  int left = 0, right = n - 1;
  
  while(left <= right) {
    int mid = left + 
              (right - left) / 2;
    if(arr[mid] == target)
      return mid;
    else if(arr[mid] < target)
      left = mid + 1;
    else
      right = mid - 1;
  }
  return -1;
}
Output (Target: 25)
Element found at index: 4

O(log n) search. Requires sorted array. Extremely efficient.

Bubble Sort C 
// Simple sorting algorithm
void bubbleSort(
    int arr[], int n) {
  for(int i = 0; i < n - 1; i++) {
    for(int j = 0; 
        j < n - i - 1; j++) {
      if(arr[j] > arr[j + 1]) {
        int temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
}
Output (Input: 64 25 12 22 11)
11 12 22 25 64

O(n²) complexity. Educational, not for production use.

Factorial (Iterative) C 
// Efficient factorial
long long factorial(
    int n) {
  if(n == 0 || n == 1)
    return 1;
  
  long long result = 1;
  for(int i = 2; 
      i <= n; i++)
    result *= i;
  
  return result;
}
Output (n=5)
120

O(n) complexity. Avoids stack overflow. Use long long.

Prime Number Check C 
// Optimized prime checker
int isPrime(int n) {
  if(n <= 1) 
    return 0;
  if(n <= 3) 
    return 1;
  if(n % 2 == 0 || 
      n % 3 == 0)
    return 0;
  
  for(int i = 5; i * i <= n; 
      i += 6)
    if(n % i == 0)
      return 0;
  return 1;
}
Output (n=17)
17 is Prime

O(√n) complexity. 6k±1 optimization.

LCM (Least Common Multiple) C 
// Least Common Multiple
int gcd(int a, int b) {
  while(b) {
    int temp = b;
    b = a % b;
    a = temp;
  }
  return a;
}
 
int lcm(int a, int b) {
  return (a * b) / gcd(a, b);
}
Output (LCM of 12, 8)
24

Uses GCD to calculate LCM. O(log min(a,b)).

Quick Sort C 
// Efficient divide-conquer sort
int partition(int arr[], 
    int low, int high) {
  int pivot = arr[high];
  int i = low - 1;
  
  for(int j = low; j < high; j++)
    if(arr[j] < pivot) {
      i++;
      int t = arr[i];
      arr[i] = arr[j];
      arr[j] = t;
    }
  arr[high] = arr[i+1];
  arr[i+1] = pivot;
  return i + 1;
}
Output (Input: 64 34 25 12)
12 25 34 64

O(n log n) average. Production-grade sorting.

Merge Sort C 
// Stable O(n log n) sort
void merge(int arr[], 
    int l, int m, int r) {
  int n1 = m - l + 1;
  int n2 = r - m;
  
  int L[n1], R[n2];
  
  for(int i = 0; i < n1; i++)
    L[i] = arr[l + i];
  for(int j = 0; j < n2; j++)
    R[j] = arr[m + 1 + j];
  
  int i = 0, j = 0, k = l;
  while(i < n1 && j < n2)
    arr[k++] = (L[i] <= R[j]) ? 
               L[i++] : R[j++];
}
Output (Guaranteed O(n log n))
12 25 34 64

Stable sorting. O(n log n) guaranteed. Uses O(n) space.

Power Function C 
// Calculate power recursively
long long power(
    long long x, 
    int y) {
  if(y == 0)
    return 1;
  
  long long res = power(x, y/2);
  
  if(y % 2 == 0)
    return res * res;
  else
    return x * res * res;
}
Output (2^10)
1024

O(log y) using binary exponentiation.

int isArmstrong(

  while(n > 0) {
    int digit = n % 10;
    sum += digit * digit * digit;

void reverseArray(

  int start = 0, end = n-1;
  while(start < end) {
    int temp = arr[start];

Insertion Sort C 
// Insertion sort algorithm
void insertionSort(
    int arr[], 
    int n) {
  for(int i = 1; i < n; i++) {
    int key = arr[i];
    int j = i - 1;
    while(j >= 0 && 
          arr[j] > key) {
      arr[j+1] = arr[j];
      j--;
    }
    arr[j+1] = key;
  }
}
Output (Input: 64 34 25 12)
12 25 34 64

O(n²) worst case, O(n) best case.

Selection Sort C 
// Selection sort algorithm
void selectionSort(
    int arr[], 
    int n) {
  for(int i = 0; i < n-1; i++) {
    int minIdx = i;
    for(int j = i+1; j < n; j++)
      if(arr[j] < arr[minIdx])
        minIdx = j;
    int temp = arr[i];
    arr[i] = arr[minIdx];
    arr[minIdx] = temp;
  }
}
Output (Input: 64 34 25 12)
12 25 34 64

O(n²) always. Simple but inefficient.
