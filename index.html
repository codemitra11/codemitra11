Palindrome Program in C

#include 
int main() {
    int n, rev = 0, temp;
    scanf("%d", &n);
    temp = n;
    while(n != 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    if(temp == rev)
        printf("Palindrome");
    else
        printf("Not Palindrome");
}

Fibonacci Series (Recursion)

#include 
int fib(int n) {
    if(n<=1) return n;
    return fib(n-1)+fib(n-2);
}
int main() {
    int i;
    for(i=0;i<10;i++)
        printf("%d ", fib(i));
}

GCD of Two Numbers

#include 
int gcd(int a, int b) {
    if(b==0) return a;
    return gcd(b, a%b);
}
int main() {
    printf("%d", gcd(12, 18));
}

Matrix Addition

#include 
int main() {
    int a[2][2], b[2][2], c[2][2];
    for(int i=0;i<2;i++)
        for(int j=0;j<2;j++)
            scanf("%d", &a[i][j]);
    for(int i=0;i<2;i++)
        for(int j=0;j<2;j++) {
            scanf("%d", &b[i][j]);
            c[i][j] = a[i][j] + b[i][j];
        }
    for(int i=0;i<2;i++) {
        for(int j=0;j<2;j++)
            printf("%d ", c[i][j]);
        printf("\n");
    }

}

Armstrong Number

    #include 
int main() {
    int n, temp, r, sum = 0;
    scanf("%d", &n);
    temp = n;

    while(n > 0) {
        r = n % 10;
        sum += r*r*r;
        n /= 10;
    }

    if(sum == temp)
        printf("Armstrong");
    else
        printf("Not Armstrong");
}

Prime Number

            #include 
int main() {
    int n, i, flag = 0;
    scanf("%d", &n);

    for(i = 2; i <= n/2; i++) {
        if(n % i == 0) {
            flag = 1;
            break;
        }
    }

    if(flag == 0 && n != 1)
        printf("Prime");
    else
        printf("Not Prime");
}
